杨辉三角

杨辉三角

原题链接

思路

根据前一行推倒后一行的数据。

class Solution(object):
    def generate(self, numRows):
        """
        :type numRows: int
        :rtype: List[List[int]]
        """
        res = list()
        for i in range(numRows):
            tmp = list()

            if i == 0:
                tmp.append(1)
            elif i == 1:
                tmp.append(1)
                tmp.append(1)
            else:
                pre = res[-1]
                tmp.append(1)
                for j in range(len(pre) - 1):
                    num = pre[j] + pre[j + 1]
                    tmp.append(num)
                tmp.append(1)

            res.append(tmp)
        return res
